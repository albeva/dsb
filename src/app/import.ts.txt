Imports:
    import { Papa } from 'ngx-papaparse';
    import { firestore } from 'firebase';

Constructor param:
    private papa: Papa,

Template:
    <p>
        <input type="file" class="btn btn-primary" (input)="upload(input)" #input>
    </p>

Functions:
    upload(input: HTMLInputElement) {
        const reader = new FileReader();
        reader.onload = (event: any) => {
            this.papa.parse(event.target.result, {
                complete: (result) => {
                    for (const item of result.data) {
                        this.addMember(item);
                    }
                }
            });
        };
        reader.readAsText(input.files[0]);
    }

    addMember(data) {
        const member = {
            character: data[0],
            legacy:    data[1],
            rankId:    this.getRankForName(data[5]),
            duty:      data[7],
            website:   data[2],
            discordId: data[4],
            joined:    firestore.Timestamp.fromDate(new Date(data[6])),
            hiatus:    data[8],
            notes:     data[9]
        };
        this.memberService.add(member);
    }

    private getRankForName(name) {
        for (const [k, v] of Object.entries(this.ranks)) {
            if (v.name === name) {
                return k;
            }
        }
    }
